# 快读模板

```c++
template<class T>
inline void get(T &res){
	char ch;
	bool bo=false;
	while(!isdigit(ch=getchar()))
		if(ch=='-')bo=true;
	res=ch-'0';
	while(isdigit(ch=getchar()))
		res=(res<<1)+(res<<3)+ch-'0';
	if(bo)res=~res+1;
	return;
}
```

```c++
template<class T>
inline void put(T x){
	if(x<0){
		putchar('-');
		x=~x+1;
	}
	if(x>9)put(x/10);
	putchar(x%10+'0');
	return;
}
```
