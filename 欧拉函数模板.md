//小于n的正整数中与n互质的数的个数
int phi(int n){
	int ans=n;
	for(int i=2;i<=n/i;i++)
		if(n%i==0){
			ans=ans/i*(i-1);
			while(n%i==0)n/=i;
		}
	if(n>1) ans=ans/n*(n-1);
	return ans;
}

//效率更高
const int N=1000005;
int primes[N],phi[N],m;
bool check[N];
inline void getphi(int n){
    phi[1] = 1;
    for(int i = 2;i <= n;i++){
        if(!check[i]){
            primes[++m] = i;
            phi[i] = i - 1;
        }
        for(int j = 1;primes[j] <= n / i;j++){
            int t = i * primes[j];
            check[t] = true;
            if(i % primes[j] == 0){
                phi[t] = phi[i] * primes[j];
                break;
            }
            phi[t] = phi[i] * (primes[j] - 1);
        }
    }
}
